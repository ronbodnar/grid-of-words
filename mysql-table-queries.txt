CREATE TABLE games(
	id BINARY(16) DEFAULT (UUID_TO_BIN(UUID())) NOT NULL PRIMARY KEY,
	word_id BINARY(16) NOT NULL,
	state ENUM('IN_PROGRESS', 'FORFEIT', 'WIN', 'LOSS') DEFAULT 'IN_PROGRESS' NOT NULL,
	start_timestamp TIMESTAMP DEFAULT CURRENT_TIMESTAMP() NOT NULL,
	end_timestamp TIMESTAMP,
	timed TINYINT(1) DEFAULT 0 NOT NULL
);

CREATE TABLE game_attempts(
    game_id BINARY(16) NOT NULL,
    attempt_number INTEGER NOT NULL,
    attempt_word VARCHAR(50) NOT NULL
);

CREATE TABLE words(
	id BINARY(16) DEFAULT (UUID_TO_BIN(UUID())) NOT NULL PRIMARY KEY,
	word VARCHAR(50) NOT NULL
);

CREATE TABLE users(
	id BINARY(16) DEFAULT (UUID_TO_BIN(UUID())) NOT NULL PRIMARY KEY,
    username VARCHAR(50) NOT NULL,
    password VARCHAR(50) NOT NULL,
    creation_date TIMESTAMP DEFAULT CURRENT_TIMESTAMP() NOT NULL
);

CREATE TABLE user_games(
	user_id BINARY(16) NOT NULL,
	game_id BINARY(16) NOT NULL
);
